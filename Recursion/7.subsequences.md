export function Subsequences(arr: number[]){
    let result: number[][] = []
    let seq: number[] = []
    getAllSubsequences(arr, 0, result, seq)
    console.log(result)

function getAllSubsequences(arr: number[], idx: number, result: number[][], seq: number[]){
    if(idx >= arr.length){
        result.push([...seq])
        return
    }

    seq.push(arr[idx])
    getAllSubsequences(arr, idx + 1, result, seq)
    seq.pop()
    getAllSubsequences(arr, idx + 1, result, seq)
}
