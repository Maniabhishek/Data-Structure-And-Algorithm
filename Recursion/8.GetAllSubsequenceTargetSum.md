function getAllSubsequnceTargetSum(arr: number[], idx: number, target: number, currentSum: number, result: number[][], seq: number[]){
    if(idx >= arr.length){
        if(currentSum === target) {
            result.push([...seq])
        }
        return
    }

    seq.push(arr[idx])
    currentSum += arr[idx]
    getAllSubsequnceTargetSum(arr, idx + 1, target, currentSum, result, seq)
    seq.pop()
    currentSum -= arr[idx]
    getAllSubsequnceTargetSum(arr, idx + 1, target, currentSum, result, seq)
}
