### Implement stack using queue

<img src="https://github.com/user-attachments/assets/5d3eb763-4e6f-4597-a2ed-a9abd99d3f51" width=400 />

```ts
class Stack{
  queue: Queue;

  push(){
    let size = queue.size()
    queue.push(x)
    for(let i = 1; i<=size; i++){
      let el = queue.top()
      queue.push(el)
      queue.pop()
    }
  }

  pop(){
    return queue.pop()
  }

  top(){
    return queue.top()
  }
}
```
