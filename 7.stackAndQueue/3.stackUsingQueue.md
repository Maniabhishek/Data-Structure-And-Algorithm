```ts
class Stack{
  queue: Queue;

  push(){
    let size = queue.size()
    queue.push(x)
    for(let i = 1; i<=size; i++){
      let el = queue.top()
      queue.push(el)
      queue.pop()
    }
  }

  pop(){
    return queue.pop()
  }

  top(){
    return queue.top()
  }
}
```
