class LNode<T> {
    value: T
    next: LNode<T>

    constructor(value: T){
        this.value = value
        this.next = null
    }
}

class StackLL<T> {
    top: LNode<T> = null
    length: number = 0;
    push(value: T){
        const newNode = new LNode(value)

        newNode.next = this.top
        this.top = newNode
        this.length += 1
    }

    pop(){
        if(this.length <= 0 ) return
        
        this.length -= 1
        const poppedElement = this.top.value
        this.top = this.top.next
        return poppedElement
    }

    topElement(){
        if(this.length === 0 ) {
            console.log(null)
            return
        }
        console.log(this.top.value)
        return this.top.value
    }

    getLength(){
        console.log(this.length)
        return this.length
    }
}

export function TestStackLL(){
    const stll = new StackLL()

    stll.push(1)
    stll.topElement() // 1

    stll.push(2)
    stll.topElement() // 2

    stll.push(3)
    stll.topElement()//3

    stll.pop()
    stll.topElement() // 2

    stll.pop()
    stll.topElement() // 2
    stll.pop()
    stll.topElement() // null
}

class QueueLL<T>{
    start: LNode<T>;
    end: LNode<T>;
    length: number = 0;

    push(value: T){
        const newNode = new LNode<T>(value)

        if(this.length === 0){
            this.start = newNode
            this.end = newNode
        }else {
            newNode.next = this.end
            this.end = newNode
        }
        this.length += 1
    }

    pop(){
        let temp = this.end
        if(temp == null){
            console.log("underflow")
            return new Error("underflow")
        }
        const poppedElement = this.start.value
        this.start = this.start.next
        this.length -= 1
        if(this.start == null) this.end = null
        return poppedElement
    }

    topElement(){
        if(this.length > 0){
            console.log(this.start.value)
            return this.start.value
        }

        console.log(null)
        return null
    }

    getLength(){
        console.log(this.length)
        return this.length
    }
}
