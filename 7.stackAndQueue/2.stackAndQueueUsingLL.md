class LNode<T> {
    value: T
    next: LNode<T>

    constructor(value: T){
        this.value = value
        this.next = null
    }
}

class StackLL<T> {
    top: LNode<T> = null
    length: number = 0;
    push(value: T){
        const newNode = new LNode(value)

        newNode.next = this.top
        this.top = newNode
        this.length += 1
    }

    pop(){
        if(this.length <= 0 ) return
        
        this.length -= 1
        const poppedElement = this.top.value
        this.top = this.top.next
        return poppedElement
    }

    topElement(){
        if(this.length === 0 ) {
            console.log(null)
            return
        }
        console.log(this.top.value)
        return this.top.value
    }

    getLength(){
        console.log(this.length)
        return this.length
    }
}

export function TestStackLL(){
    const stll = new StackLL()

    stll.push(1)
    stll.topElement() // 1

    stll.push(2)
    stll.topElement() // 2

    stll.push(3)
    stll.topElement()//3

    stll.pop()
    stll.topElement() // 2

    stll.pop()
    stll.topElement() // 2
    stll.pop()
    stll.topElement() // null
}
